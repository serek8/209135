271  AVLTreeNode<ContentType> * remove(AVLTreeNode<ContentType> * x)
272  {
273  AVLTreeNode<ContentType> * t, * y, * z;
274  bool nest;
275 
276  // Jeśli węzeł x posiada dwójkę dzieci, lewego i prawego potomka:
277  if((x->leftNode) && (x->rightNode))

282  }

284  else {
285  if(x->leftNode) {

291  }

293  nest = true;
294  }
295 
296  if(y) {
297  y->parentNode = x->parentNode;
298  if(x->leftNode)
299  {
300  y->leftNode = x->leftNode;
301  y->leftNode->parentNode = y;
302  }
303  if(x->rightNode)
304  {
305  y->rightNode = x->rightNode;
306  y->rightNode->parentNode = y;
307  }
308  y->balanceFactor = x->balanceFactor;
309  }
310 
311  if(x->parentNode) {
312  if(x->parentNode->leftNode == x) x->parentNode->leftNode = y; else x->parentNode->rightNode = y;
313  }
314  else rootNode = y;
315 
316  if(nest) {
317  z = y;
318  y = x->parentNode;
319  while(y)
320  {
321  // węzeł y był w stanie równowagi przed usunięciem węzła x w jednym z jego poddrzew.
322  if(!(y->balanceFactor)) {

324  break;
325  }
326  else {
327  //skrócone zostało cięższe poddrzewo
328  if (((y->balanceFactor == 1) &&
329  (y->leftNode == z)) || ((y->balanceFactor == -1) &&
330  (y->rightNode == z))) {
331  y->balanceFactor = 0;
332  z = y; y = y->parentNode;
333  }
334  else {

336 
337  //Wykonujemy odpowiednią rotację pojedynczą
338  if(!(t->balanceFactor)) {

341  }
342  //Wykonujemy odpowiednią rotację pojedynczą
343  else if(y->balanceFactor == t->balanceFactor)
